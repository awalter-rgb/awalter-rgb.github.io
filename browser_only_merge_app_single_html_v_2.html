"""PDF Merger Web App (Flask) What it does - Accepts uploads (.pdf, .docx, .pptx) - Converts .docx and .pptx to PDF using LibreOffice (soffice) in headless mode - Merges everything into ONE PDF and returns it as a download Run pip install flask pypdf python app.py Then open: http://127.0.0.1:5000 Notes - Requires LibreOffice to be installed and the 'soffice' executable to be on PATH. - Upload order is preserved in the order the browser submits the files. """ from __future__ import annotations import os import re import shutil import subprocess import tempfile from pathlib import Path from flask import Flask, Response, flash, redirect, render_template_string, request, send_file, url_for from pypdf import PdfReader, PdfWriter APP = Flask(__name__) APP.secret_key = os.environ.get("APP_SECRET", "change-me") # Adjust as you like (bytes). 200 MB default. APP.config["MAX_CONTENT_LENGTH"] = int(os.environ.get("MAX_UPLOAD_BYTES", str(200 * 1024 * 1024))) ALLOWED_EXTS = {".pdf", ".docx", ".pptx"} HTML = """ Merge PDFs, DOCX & PPTX into one PDF

Merge PDFs, DOCX, and PPTX files into one PDF

{% with messages = get_flashed_messages(with_categories=true) %} {% if messages %}
{% for category, message in messages %}
{{ message }}
{% endfor %}
{% endif %} {% endwith %}

Tip: the merge order follows the upload order. If you want a specific order, select files in that order (or rename them 01_, 02_, etc.).

"".pdf" will be added if you omit it.
What happens to DOCX / PPTX files?
  • They are converted to PDF with LibreOffice in headless mode.
  • All PDFs are then merged into a single PDF.
""" def _safe_filename(name: str) -> str: name = (name or "merged.pdf").strip() name = re.sub(r"[^A-Za-z0-9._ -]+", "_", name) if not name.lower().endswith(".pdf"): name += ".pdf" # Avoid empty / weird return name if name not in {".pdf", ""} else "merged.pdf" def _soffice_convert_to_pdf(in_path: Path, out_dir: Path) -> Path: """Convert a DOCX or PPTX to PDF using LibreOffice (soffice).""" cmd = [ "soffice", "--headless", "--nologo", "--nolockcheck", "--nodefault", "--norestore", "--convert-to", "pdf", "--outdir", str(out_dir), str(in_path), ] proc = subprocess.run(cmd, stdout=subprocess.PIPE, stderr=subprocess.STDOUT, text=True) if proc.returncode != 0: raise RuntimeError(f"LibreOffice conversion failed for {in_path.name}. Output: {proc.stdout}") pdf_path = out_dir / (in_path.stem + ".pdf") if not pdf_path.exists(): # LibreOffice sometimes changes capitalization or uses slightly different naming. # Fall back to the newest PDF in out_dir. pdfs = sorted(out_dir.glob("*.pdf"), key=lambda p: p.stat().st_mtime, reverse=True) if not pdfs: raise RuntimeError(f"LibreOffice reported success but produced no PDF for {in_path.name}.") pdf_path = pdfs[0] return pdf_path def _merge_pdfs(pdf_paths: list[Path], out_pdf: Path) -> None: writer = PdfWriter() for p in pdf_paths: reader = PdfReader(str(p)) for page in reader.pages: writer.add_page(page) with out_pdf.open("wb") as f: writer.write(f) @APP.get("/") def index() -> str: return render_template_string(HTML) @APP.post("/merge") def merge() -> Response: files = request.files.getlist("files") if not files: flash("Please upload at least one file.", "error") return redirect(url_for("index")) outname = _safe_filename(request.form.get("outname", "merged.pdf")) # Work in an isolated temp directory and clean up after. with tempfile.TemporaryDirectory(prefix="merge_upload_") as td: td_path = Path(td) in_dir = td_path / "in" conv_dir = td_path / "converted" in_dir.mkdir(parents=True, exist_ok=True) conv_dir.mkdir(parents=True, exist_ok=True) pdfs_to_merge: list[Path] = [] for idx, storage in enumerate(files, start=1): orig_name = storage.filename or f"file_{idx}" ext = Path(orig_name).suffix.lower() if ext not in ALLOWED_EXTS: flash(f"Unsupported file type: {orig_name}. Allowed: PDF, DOCX.", "error") return redirect(url_for("index")) # Save with an index prefix to preserve ordering and avoid collisions. safe_base = re.sub(r"[^A-Za-z0-9._ -]+", "_", Path(orig_name).stem) or f"file_{idx}" saved_path = in_dir / f"{idx:02d}_{safe_base}{ext}" storage.save(saved_path) if ext == ".pdf": pdfs_to_merge.append(saved_path) else: # DOCX/PPTX -> PDF pdf_path = _soffice_convert_to_pdf(saved_path, conv_dir) # Copy into a stable name so merge order is obvious. stable_pdf = conv_dir / f"{idx:02d}_{safe_base}.pdf" shutil.copyfile(pdf_path, stable_pdf) pdfs_to_merge.append(stable_pdf) out_pdf = td_path / "merged.pdf" _merge_pdfs(pdfs_to_merge, out_pdf) # Stream back to the user. return send_file( out_pdf, as_attachment=True, download_name=outname, mimetype="application/pdf", ) if __name__ == "__main__": # Set host="0.0.0.0" if you want LAN access. APP.run(host="127.0.0.1", port=int(os.environ.get("PORT", "5000")), debug=True)
